BASE DE DATOS (SCHEMA) (
	Estructura / almacen de datos
	Compuestas por Tablas(organizador tematico), Campos(columnas) y registros(filas)
	TIPOS (
		RELACIONAL(*),
		JERARQUICA,
		TRANSACCIONAL,
		ORIENTADA A OBJETOS
		ETC...
	)
)
Motor Base Datos(MySQL Workbench)(
	Desarrollado por IBM (SQL)
	Adquirido por ORACLE (MySQL, Oracle SQL)
	Proyecto OpenSource (MariaDB)
)

SQL (Structured Query Language)(
	DDL(Data Definition Language)
		CREATE	crear bd/tablas/usuarios
		DROP	eliminar bd/tablas/usuarios
		MODIFY	modificar tablas

	DML(Data Manipulation Language)
		INSERT	insertar registros
		DELETE	eliminar registros
		UPDATE	actualizar registros
		SELECT	consultar registros

	DCL(Data Control Language)
		GRANT	otorgar permisos
		REVOKE	quitar permisos
		FLUSH PRIVILEGES actualizar permisos
)
