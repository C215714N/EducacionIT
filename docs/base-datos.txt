SQL(Structured Query Language)(
	DDL(Data Definition Language)(
		CREATE: crear db | tablas | usuarios
		DROP:	elimina db | tablas | usuarios
		ALTER: 	modifica tablas
	),
	DML(Data Manipulation Language)(
		INSERT: agrega registros a una tabla
		DELETE: elimina registros de una tabla
		UPDATE: actualiza registros de una tabla
		SELECT: consulta los registro de una tabla
	),
	DCL(Data Control Language)(
		GRANT:	otorga permisos 
		REVOKE: quita permisos
		FLUSH PRIVILEGES: actualiza los permisos
	)
);
