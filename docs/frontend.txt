HTML (HyperText Markup Language){
	Lenguaje de Marcado de Hipertexto
	ESTRUCTURA => Utiliza Etiquetas
	Ejemplo:
		<p> soy un parrado </p>
		"El Orden de los Factores ALTERA el Producto" (Esqueleto)
}
CSS (Cascading Styles Sheets){
	Hojas de Estilo en Cascada
	ESTILOS => Utiliza Reglas
	Ejemplo:
		p{ color:blue; }
		"El Orden de los Factores NO ALTERA el Producto" (Apariencia)
}
