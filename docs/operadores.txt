OPERADORES MATEMATICOS
	+	suma
	-	resta
	*	multiplicacion
	/	division
	%	modulo

OPERADORES DE COMPARACION
	=	igual a
	>	mayor que
	>=	mayor o igual
	<	menor que
	<=	menor o igual
	!=	no es igual
	<>	distinto a

OPERADORES LOGICOS
	AND		si "a" y "b" 	entonces	true 
	NOT		si no "a" 	entonces	true
	OR 		si "a" o "b" 	entonces	true 
	XOR		si "a" y no "b"	entonces	true

OPERADORES DE FUNCION
	LIKE '%a%'		evalua si existe valor similar
	IN(a,b)			evalua si alguno de los valores existe
	BETWEEN a AND b evalua si un valor se encuentra en medio
